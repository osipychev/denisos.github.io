dict_of_pieces = {'p': (0, -1), 'P': (0, 1), # pawn
                      'r': (1, -1), 'R': (1, 1), # rook
                      'n': (2, -1), 'N': (2, 1), # knight
                      'b': (3, -1), 'B': (3, 1), # bishop
                      'q': (4, -1), 'Q': (4, 1), # queen
                      'k': (5, -1), 'K': (5, 1)} # king

# transcribing chess board into multichannel sample
def board2sample(board):
    
    # not taking turn into account
    #sample = np.zeros((8, 8, 6)) 
    
    # taking turn into prediction
    sample = np.zeros((8, 8, 7)) 
    if board.turn: sample[:,:,-1] = 1
    else: sample[:,:,-1] = -1
    
    board_str = str(board).replace(" ", "").replace("\n", "").replace(".", "0")
    for row_idx in range(8):
        for col_idx in range(8):
            piece = board_str[row_idx*8+col_idx]
            if piece == '0':
                continue
            attr = dict_of_pieces[piece]
            sample[row_idx, col_idx, attr[0]] = attr[1]
    return sample

# transcribing multichannel sample back to chess board
def sample2board(s):
    
    sample = s.astype(int)
    
    val_list = list(dict_of_pieces.values())
    keys_list = list(dict_of_pieces.keys())
    
    board_fen = ""
    k = 0
    for row_idx in range(8):
        for col_idx in range(8):
            piece = None
            
            for chn_idx in range(6):
                color = sample[row_idx, col_idx, chn_idx]
                if color != 0:
                    idx = val_list.index((chn_idx, color))
                    piece = keys_list[idx]
                    break
            
            if piece is None:
                k += 1
                continue
            
            if k != 0:
                board_fen += str(k)
                k = 0
            board_fen += str(piece)
                
        if k != 0: 
            board_fen += str(k)
            k = 0
        board_fen += "/"
               
    return chess.Board(board_fen[:-1])

def read_games(pgn_file, hdf5_file, chunk_size=1000, shuffle=False):
    
    X_buffer, Y_buffer = [], []
    
    pgn = open(pgn_file)
    hdf = h5py.File(hdf5_file, 'a')
    hdf.create_dataset("data", shape=(0,8,8,7), maxshape=(None, 8, 8, 7), dtype='int')
    hdf.create_dataset("label", shape=(0, 1), maxshape=(None, 1), dtype='int')
    
    k = 0
    while True:
        
        try:
            game = chess.pgn.read_game(pgn)
        except:
            break
        
        if game is None: break

        result = game.headers['Result']
        
        # skip the games that ended a draw
        if result =='1/2-1/2': continue
        
        # start a new board and make step by step
        board = game.board()
        for move in game.mainline_moves():
            k += 1
            board.push(move)
            
            X_buffer.append(board2sample(board)) # transcribe the board into channels
            if result == '1-0': Y_buffer.append(1)  # white win
            elif result == '0-1': Y_buffer.append(0) # white lose
                
            if (k % chunk_size == 0):
                X_np = np.asarray(X_buffer)
                Y_np = np.expand_dims(np.asarray(Y_buffer), axis=1)
                
                indices = np.arange(chunk_size)
                
                #!!! shuffling is done inside the chunk only, so use very large chunks!!!
                if shuffle: np.random.shuffle(indices)
                
                # expand database and append data
                hdf["data"].resize((hdf["data"].shape[0] + X_np.shape[0]), axis = 0)
                hdf["label"].resize((hdf["label"].shape[0] + Y_np.shape[0]), axis = 0)
                hdf["data"][-X_np.shape[0]:] = X_np[indices,:]
                hdf["label"][-Y_np.shape[0]:] = Y_np[indices,:]
                
                X_buffer, Y_buffer = [], []
                
                print("Boards parsed from the dataset: ", hdf["data"].shape[0], ".", end="\r", flush=True)
                
    pgn.close()
    hdf.close()
    return True

3. Create a simple TF Keras model¶

I start with a simple Keras model requiring minimum effort in order to understand if the data worth making a model. The process is simple, load data into a memory, feed it to a function approximator all at once, evaluate the result on the test set.

from tensorflow import keras

model = keras.models.Sequential([
    keras.layers.Conv2D(filters=64, kernel_size=(5,5), padding='same', activation='relu'),
    keras.layers.Dropout(0.1),
    keras.layers.Conv2D(filters=64, kernel_size=(3,3), padding='valid', activation='relu'),
    keras.layers.Flatten(),
    keras.layers.Dense(1, activation='sigmoid')
])

model.compile(loss='binary_crossentropy', optimizer=keras.optimizers.Adam(lr=0.001), metrics=['accuracy'])
history = model.fit(train_x, train_y, batch_size=10000, epochs=10, shuffle=True)

5. Summary¶

The average prediction accuracy is about 68% which is not a lot. However, this accuracy uses a wrong evaluation metrics. The boards with no clear leader, for instance every begining of the game, are still evoluated as either white or black win. On average, the prediction for this game is 0.5-0.5 which is correct.
