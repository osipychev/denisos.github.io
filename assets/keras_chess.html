dict_of_pieces = {'p': (0, -1), 'P': (0, 1), # pawn
                      'r': (1, -1), 'R': (1, 1), # rook
                      'n': (2, -1), 'N': (2, 1), # knight
                      'b': (3, -1), 'B': (3, 1), # bishop
                      'q': (4, -1), 'Q': (4, 1), # queen
                      'k': (5, -1), 'K': (5, 1)} # king

# transcribing chess board into multichannel sample
def board2sample(board):
    
    # not taking turn into account
    #sample = np.zeros((8, 8, 6)) 
    
    # taking turn into prediction
    sample = np.zeros((8, 8, 7)) 
    if board.turn: sample[:,:,-1] = 1
    else: sample[:,:,-1] = -1
    
    board_str = str(board).replace(" ", "").replace("\n", "").replace(".", "0")
    for row_idx in range(8):
        for col_idx in range(8):
            piece = board_str[row_idx*8+col_idx]
            if piece == '0':
                continue
            attr = dict_of_pieces[piece]
            sample[row_idx, col_idx, attr[0]] = attr[1]
    return sample

# transcribing multichannel sample back to chess board
def sample2board(s):
    
    sample = s.astype(int)
    
    val_list = list(dict_of_pieces.values())
    keys_list = list(dict_of_pieces.keys())
    
    board_fen = ""
    k = 0
    for row_idx in range(8):
        for col_idx in range(8):
            piece = None
            
            for chn_idx in range(6):
                color = sample[row_idx, col_idx, chn_idx]
                if color != 0:
                    idx = val_list.index((chn_idx, color))
                    piece = keys_list[idx]
                    break
            
            if piece is None:
                k += 1
                continue
            
            if k != 0:
                board_fen += str(k)
                k = 0
            board_fen += str(piece)
                
        if k != 0: 
            board_fen += str(k)
            k = 0
        board_fen += "/"
               
    return chess.Board(board_fen[:-1])

Parse the dataset¶

Here, we parse the games with a non-draw score. Esentially, we want to learn a proximity to the terminal state.

%%time

def read_games(pgn_file, hdf5_file, gamma=0.9):
    
    pgn = open(pgn_file)
    hdf = h5py.File(hdf5_file, 'a')
    hdf.create_dataset("data", shape=(0,8,8,7), maxshape=(None, 8, 8, 7), dtype='int')
    hdf.create_dataset("label", shape=(0, 1), maxshape=(None, 1), dtype='float')
    
    k_boards = 0
    k_games = 0
    while True:
        
        k_games += 1
        
        try:
            game = chess.pgn.read_game(pgn)
        except:
            break
        
        if game is None: break

        result = game.headers['Result']
        
        # skip the games that ended with a draw
        if result =='1/2-1/2': continue
        
        # start a new board and make step by step
        board = game.board()
        X_buffer, Y_buffer = [], []
        
        if result == '1-0': Y_buffer.append(1)  # white win
        elif result == '0-1': Y_buffer.append(-1) # white lose
        else: continue
        
        for move in game.mainline_moves():
            k_boards += 1
            board.push(move)
            
            X_buffer.append(board2sample(board)) # transcribe the board into channels
            Y_buffer.insert(0, gamma * Y_buffer[0])
                
        X_np = np.asarray(X_buffer)
        Y_np = np.expand_dims(np.asarray(Y_buffer[1:]), axis=1)

        # expand database and append data
        hdf["data"].resize((hdf["data"].shape[0] + X_np.shape[0]), axis = 0)
        hdf["label"].resize((hdf["label"].shape[0] + Y_np.shape[0]), axis = 0)
        hdf["data"][-X_np.shape[0]:] = X_np
        hdf["label"][-Y_np.shape[0]:] = Y_np

        X_buffer, Y_buffer = [], []

        print("Games parsed: %d. Boards parsed: %d." % (k_games,hdf["data"].shape[0]), end="\r", flush=True)
               
    result = (hdf["data"].shape[0] == hdf["label"].shape[0])
            
    pgn.close()
    hdf.close()
    return result

3. Create a simple TF Keras model¶

I start with a simple Keras model requiring minimum effort in order to understand if the data worth making a model.

Here, we want to train a regression function that will predict an expected reward (discounter reward).

In contrast with binary classification problem (predict win or loose), the regression will estimate the proximity to the terminal state reward (either to -1 or +1 depending on the end of the game). Therefore, we will use Tanh activation function and Mean Squared Error loss function.

In addition, we will be using a CNN net with 5x5x64, 3x3x64 kernels to extract features from the board. The training will be done by Adam algorithm.

Depending on the discount factor Gamma used in preprocessing and number of states in the game, the value of the initial state may vary. Often, it may fluctuate from -0.5 to 0.5. This will introduce an observation noise that results in a consistent error during the training. There is nothing we can do about it. However, the mean prediction for the initial state should be still valid assuming the process is random.

from tensorflow import keras

model = keras.models.Sequential([
    keras.layers.Conv2D(filters=32, kernel_size=(5,5), padding='same', activation='relu'),
    keras.layers.Dropout(0.5),
    keras.layers.Conv2D(filters=32, kernel_size=(3,3), padding='valid', activation='relu'),
    keras.layers.Flatten(),
    keras.layers.Dense(128, activation='relu'),
    keras.layers.Dense(1, activation='tanh')
])

model.compile(loss='mean_squared_error', optimizer=keras.optimizers.Adam(lr=0.001),
              decay=1e-3 / 100, metrics=['mae'])
history = model.fit(train_x, train_y, batch_size=10000, epochs=10, shuffle=True)
